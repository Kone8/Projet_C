#include <stdio.h>
#include "cdataframe.h"

int main() {
    // On crée un CDataFrame avec x colonnes (pour la beta, ca sera 3 en exemple)
    CDataFrame* cdf = create_cdataframe(3);

    // On défini des titres de colonnes
    char* titles[] = {"Column 1", "Column 2", "Column 3"};

    // On défini les données
    int data1[] = {1, 2, 3 };
    char data2[] = {'a', 'b', 'c'};
    float data3[] = {1.1, 2.2, 3.3};

    // On ajoute les données au Cdataframe
    for (int i = 0; i < 3; ++i) {
        cdf->columns[i] = create_column(titles[i]);
        cdf->columns[i]->num_rows = 3;
    }
    cdf->columns[0]->data = data1;
    cdf->columns[0]->type = 0; // INT
    cdf->columns[1]->data = data2;
    cdf->columns[1]->type = 1;
    cdf->columns[2]->data = data3;
    cdf->columns[2]->type = 2; // FLOAT

    // On affiche le CDataFrame
    printf("Le CDataFrame a ete cree avec succes.\n" "Nombre de colonnes : %d \n", cdf->num_columns);
    display_cdataframe(cdf);


    return 0;
}
